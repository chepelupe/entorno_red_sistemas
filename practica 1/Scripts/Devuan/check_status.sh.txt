#script check_status.sh - informacion del sistema, nombre, direccion ip y espacio total en el disco
echo ""
echo ">>> INFORMACION-SISTEMA <<<"
echo "Nombre del host: $(hostname)"
echo ""
echo ">>>DIRECCIONES IP<<<"
ip -4 addr show | grep inet | awk '{print " -" $2' | sed '/\/.*//'
echo ""
echo ">>>ESPACIO TOTAL DISPONIBLE EN EL DISCO<<<"
df -h /
echo ""
